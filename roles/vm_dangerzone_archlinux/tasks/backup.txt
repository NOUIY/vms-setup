---
- name: Create tempdir for dangerzone files
  become_user: "{{ create_user }}"
  tempfile: state=directory suffix="dangerzone"
  register: dangerzone_temp_dir  

- name: Download dangerzone archive
  become_user: "{{ create_user }}"
  get_url:
    url: "{{ github_release_url_dangerzone }}"
    dest: "{{ dangerzone_temp_dir }}/dangerzone.tar"

- name: Extract dangerzone archive
  become_user: "{{ create_user }}"
  unarchive:
    src: "{{ dangerzone_temp_dir }}/dangerzone.tar"
    dest: "{{ dangerzone_temp_dir }}"
  
- name: "Copy dangerzone binaries to {{ custom_binaries_folder }}"
  become: yes
  copy:
    src: "{{ github_perso_temp_folders.results | selectattr('item', '==', 'rclip') | map(attribute='path') | first }}/{{ root_archive_bin_path_rclip }}"
    dest: "/home/{{ create_user }}/Binaries"
    mode: "a+x"
    
- name: Create iptables file
  become_user: "{{ create_user }}"
  template:
    src: "{{ role_path }}/templates/dangerzone-httpserver.iptables.sh.j2"
    dest: "{{ dangerzone_temp_dir.path}}/dangerzone-httpserver.iptables"
    mode: "a+x"  

- name: Restrict network connectivity via iptables
  become: yes
  shell: "sh {{ dangerzone_httpserver_iptables_file | quote }}/dangerzone-httpserver.iptables"

- name: Install dangerzone-httpserver service files
  become: yes
  template: src="{{ role_path }}/templates/dangerzone-httpserver.service.j2" dest="/etc/systemd/system/dangerzone-httpserver.service"

- name: Enable dangerzone-httpserver service
  become: yes
  service:
    name: dangerzone-httpserver
    enabled: yes
